CREATE DATABASE dms;
USE dms;
-- Admins table
CREATE TABLE `admins` (
    `AdminID` INT AUTO_INCREMENT PRIMARY KEY,
    `Email` VARCHAR(255) UNIQUE NOT NULL,
    `Password` VARCHAR(255) NOT NULL
);

-- Volunteers table
CREATE TABLE `volunteers` (
    `VolunteerID` INT AUTO_INCREMENT PRIMARY KEY,
    `Name` VARCHAR(100) NOT NULL,
    `Age` INT NOT NULL,
    `Gender` ENUM('Male', 'Female', 'Other') NOT NULL,
    `CNIC` VARCHAR(15) UNIQUE NOT NULL,
    `Contact` VARCHAR(15) NOT NULL,
    `Email` VARCHAR(100) UNIQUE NOT NULL,
    `Password` VARCHAR(255) NOT NULL,
    `Address` TEXT NOT NULL
);

-- Volunteer skills (1:M relationship for volunteers and skills)
CREATE TABLE `volunteer_skills` (
    `SkillID` INT AUTO_INCREMENT PRIMARY KEY,
    `VolunteerID` INT NOT NULL,
    `Skill` VARCHAR(100) NOT NULL,
    FOREIGN KEY (`VolunteerID`) REFERENCES `volunteers`(`VolunteerID`) ON DELETE CASCADE
);

-- Disaster zones table
CREATE TABLE `disasterzones` (
    `ZoneID` INT AUTO_INCREMENT PRIMARY KEY,
    `DisasterName` VARCHAR(255) NOT NULL,
    `Description` TEXT,
    `Location` VARCHAR(255) NOT NULL,
    `LastUpdated` TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- Victims table
CREATE TABLE `victims` (
    `VictimID` INT AUTO_INCREMENT PRIMARY KEY,
    `Name` VARCHAR(100) NOT NULL,
    `Age` INT,
    `Gender` ENUM('Male', 'Female', 'Other'),
    `Location` VARCHAR(255) NOT NULL,
    `Needs` TEXT,
    `ContactInfo` VARCHAR(50),
    `ZoneID` INT NOT NULL,
    `LastUpdated` TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    FOREIGN KEY (`ZoneID`) REFERENCES `disasterzones`(`ZoneID`) ON DELETE CASCADE
);

-- Resources table
CREATE TABLE `resources` (
    `ResourceID` INT AUTO_INCREMENT PRIMARY KEY,
    `ResourceType` VARCHAR(100) NOT NULL,
    `Quantity` INT NOT NULL,
    `Location` VARCHAR(255),
    `LastUpdated` TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- Donations table
CREATE TABLE `donations` (
    `DonationID` INT AUTO_INCREMENT PRIMARY KEY,
    `DonorName` VARCHAR(100),
    `AccountNO` VARCHAR(13),
    `Amount` DECIMAL(10,2) NOT NULL,
    `DonationDate` DATE DEFAULT CURRENT_DATE
);

-- Volunteer assignments
CREATE TABLE `volunteerassignments` (
    `AssignmentID` INT AUTO_INCREMENT PRIMARY KEY,
    `VolunteerID` INT NOT NULL,
    `ZoneID` INT NOT NULL,
    `TaskDescription` TEXT NOT NULL,
    `AssignmentDate` TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (`VolunteerID`) REFERENCES `volunteers`(`VolunteerID`) ON DELETE CASCADE,
    FOREIGN KEY (`ZoneID`) REFERENCES `disasterzones`(`ZoneID`) ON DELETE CASCADE
);

-- Volunteer assignment log
CREATE TABLE `volunteerassignmentlog` (
    `LogID` INT AUTO_INCREMENT PRIMARY KEY,
    `AssignmentID` INT NOT NULL,
    `VolunteerID` INT NOT NULL,
    `ZoneID` INT NOT NULL,
    `Action` VARCHAR(255),
    `Date` TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (`AssignmentID`) REFERENCES `volunteerassignments`(`AssignmentID`) ON DELETE CASCADE,
    FOREIGN KEY (`VolunteerID`) REFERENCES `volunteers`(`VolunteerID`) ON DELETE CASCADE,
    FOREIGN KEY (`ZoneID`) REFERENCES `disasterzones`(`ZoneID`) ON DELETE CASCADE
);
